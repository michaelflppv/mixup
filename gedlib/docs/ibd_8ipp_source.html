37 template<>
38 IBD<GXLLabel, GXLLabel>::
39 IBD(const std::string & otu_distances, double alpha) :
40 node_rel_costs_(),
41 alpha_{alpha},
42 otus_() {
43  std::ifstream csv_file(otu_distances);
44  std::string row;
45  std::getline(csv_file, row);
46  std::vector<std::string> row_as_vector;
47  util::tokenize(row, ',', row_as_vector);
48  std::size_t max_otu{0};
49  std::vector<std::string> otu_as_vector;
50  for (const std::string & otu_with_prefix : row_as_vector) {
51  otu_as_vector.clear();
52  util::tokenize(otu_with_prefix, '_', otu_as_vector);
53  otus_.emplace_back(std::stoul(otu_as_vector.at(1)));
54  max_otu = std::max(max_otu, otus_.back());
55  }
56  node_rel_costs_.resize(max_otu + 1, max_otu + 1);
57  node_rel_costs_.set_to_val(0);
58 
59  std::size_t otu_1;
60  std::size_t otu_2;
61  while(std::getline(csv_file, row)) {
62  row_as_vector.clear();
63  util::tokenize(row, ',', row_as_vector);
64  otu_as_vector.clear();
65  util::tokenize(row_as_vector.at(0), '_', otu_as_vector);
66  otu_1 = std::stoul(otu_as_vector.at(1));
67  for (std::size_t pos{1}; pos < row_as_vector.size(); pos++) {
68  otu_2 = otus_.at(pos - 1);
69  node_rel_costs_(otu_1, otu_2) = std::stod(row_as_vector.at(pos));
70  }
71  }
72  node_rel_costs_ /= node_rel_costs_.max();
73 }

96 template<>
97 GXLLabel
98 IBD<GXLLabel, GXLLabel>::
99 median_node_label(const std::vector<GXLLabel> & node_labels) const {
100  // Transform labels to indices.
101  std::vector<std::size_t> otus;
102  for (const auto & label : node_labels) {
103  otus.emplace_back(std::stoul(label.at("OTU")));
104  }
105 
106  // Determine the feature with the smallest SOD to all node features.
107  std::size_t best_otu{undefined()};
108  double best_sod{std::numeric_limits<double>::infinity()};
109  double current_sod{0.0};
110  for (auto otu_1 : otus_) {
111  current_sod = 0.0;
112  for (auto otu_2 : otus) {
113  current_sod += node_rel_costs_(otu_1, otu_2);
114  }
115  if (current_sod < best_sod) {
116  best_sod = current_sod;
117  best_otu = otu_1;
118  }
119  }
120 
121  // Construct and return the median label.
122  ged::GXLLabel median_label;
123  median_label["OTU"] = std::to_string(best_otu);
124  return median_label;
125 }

150 template<>
151 GXLLabel
152 IBD<GXLLabel, GXLLabel>::
153 median_edge_label(const std::vector<GXLLabel> & edge_labels) const {
154 
155  // Collect normalized log-ratios.
156  std::vector<double> nlogratios;
157  for (const auto & label : edge_labels) {
158  nlogratios.emplace_back(std::stod(label.at("nlogratio")));
159  }
160 
161  // Compute the median.
162  double median;
163  const auto middle_itr = nlogratios.begin() + nlogratios.size() / 2;
164  std::nth_element(nlogratios.begin(), middle_itr, nlogratios.end());
165  if (nlogratios.size() % 2 == 0) {
166  const auto left_middle_itr = std::max_element(nlogratios.begin(), middle_itr);
167  median = (*left_middle_itr + *middle_itr) / 2.0;
168  }
169  else {
170  median = *middle_itr;
171  }
172 
173  // Construct and return the median label.
174  ged::GXLLabel median_label;
175  median_label["nlogratio"] = std::to_string(median);
176  return median_label;
177 }
