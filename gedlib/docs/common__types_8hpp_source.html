405 template<class UserNodeID, class UserNodeLabel, class UserEdgeLabel>
406 std::ostream & operator<<(std::ostream & os, const ExchangeGraph<UserNodeID, UserNodeLabel, UserEdgeLabel> & graph) {
407  os << "graph [\n";
408  os << "\tid = " << graph.id << "\n";
409  os << "\tdirected = 0\n";
410  os << "\tnum_nodes = " << graph.num_nodes << "\n";
411  os << "\tnum_edges = " << graph.num_edges << "\n";
412  for (std::size_t node_id{0}; node_id < graph.num_nodes; node_id++) {
413  os << "\tnode [\n";
414  os << "\t\tinternal_id = " << node_id << "\n";
415  os << "\t\toriginal_id = \"" << graph.original_node_ids.at(node_id) << "\"\n";
416  os << "\t\tlabel = \"" << graph.node_labels.at(node_id) << "\"\n";
417  os << "\t]\n";
418  }
419  for (std::size_t tail_id{0}; tail_id < graph.num_nodes; tail_id++) {
420  for (std::size_t head_id{tail_id + 1}; head_id < graph.num_nodes; head_id++) {
421  if (graph.adj_matrix[tail_id][head_id] == 1) {
422  os << "\tedge [\n";
423  os << "\t\tinternal_id_tail = " << tail_id << "\n";
424  os << "\t\tinternal_id_head = " << head_id << "\n";
425  os << "\t\tlabel = \"" << graph.edge_labels[std::make_pair(tail_id, head_id)] << "\"\n";
426  os << "\t]\n";
427  }
428  }
429  }
430  os << "]\n";
431  return os;
432 }
433 #endif /* ENABLE_GRAPH_STREAMING */
