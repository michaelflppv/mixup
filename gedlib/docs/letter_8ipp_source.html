58 template<>
59 double
60 Letter<GXLLabel, GXLLabel>::
61 node_rel_cost_fun(const GXLLabel & node_label_1, const GXLLabel & node_label_2) const {
62  double x_l_minus_x_r(std::stod(node_label_1.at("x")) - std::stod(node_label_2.at("x")));
63  double y_l_minus_y_r(std::stod(node_label_1.at("y")) - std::stod(node_label_2.at("y")));
64  double squared_euclidean{std::pow(x_l_minus_x_r, 2) + std::pow(y_l_minus_y_r, 2)};
65  if (euclidean_exponent_ == 2) {
66  return alpha_ * squared_euclidean;
67  }
68  if (euclidean_exponent_ == 1) {
69  return alpha_ * std::sqrt(squared_euclidean);
70  }
71  return alpha_ * std::pow(squared_euclidean, euclidean_exponent_ / 2);
72 }

74 template<>
75 GXLLabel
76 Letter<GXLLabel, GXLLabel>::
77 median_node_label(const std::vector<GXLLabel> & node_labels) const {
78  // Transform the labels into two-dimensional coordinates and compute mean label as initial solution.
79  std::vector<std::pair<double, double>> node_labels_as_coords;
80  double label_x, label_y;
81  double sum_x{0};
82  double sum_y{0};
83  for (const auto & node_label : node_labels) {
84  label_x = std::stod(node_label.at("x"));
85  label_y = std::stod(node_label.at("y"));
86  sum_x += label_x;
87  sum_y += label_y;
88  node_labels_as_coords.emplace_back(label_x, label_y);
89  }
90  std::pair<double, double> median(sum_x / static_cast<double>(node_labels.size()), sum_y / static_cast<double>(node_labels.size()));
91  if (euclidean_exponent_ == 2) {
92  ged::GXLLabel median_label;
93  median_label["x"] = std::to_string(median.first);
94  median_label["y"] = std::to_string(median.second);
95  return median_label;
96  }
97 
98  // Run main loop of Weiszfeld's Algorithm.
99  double epsilon{0.0001};
100  double delta{1.0};
101  std::size_t num_itrs{0};
102  bool all_equal{false};
103  while ((delta > epsilon) and (num_itrs++ < 100) and (not all_equal)) {
104  std::pair<double, double> numerator(0, 0);
105  double denominator{0};
106  for (const auto & node_label_as_coord : node_labels_as_coords) {
107  double norm{0};
108  norm += (node_label_as_coord.first - median.first) * (node_label_as_coord.first - median.first);
109  norm += (node_label_as_coord.second - median.second) * (node_label_as_coord.second - median.second);
110  norm = std::sqrt(norm);
111  if (norm > 0) {
112  numerator.first += node_label_as_coord.first / norm;
113  numerator.second += node_label_as_coord.second / norm;
114  denominator += 1.0 / norm;
115  }
116  }
117  if (denominator == 0) {
118  all_equal = true;
119  }
120  else {
121  std::pair<double, double> new_median(numerator.first / denominator, numerator.second / denominator);
122  delta = std::abs(median.first - new_median.first) + std::abs(median.second - new_median.second);
123  median = new_median;
124  }
125  }
126 
127  // Transform the solution to ged::GXLLabel and return it.
128  ged::GXLLabel median_label;
129  median_label["x"] = std::to_string(median.first);
130  median_label["y"] = std::to_string(median.second);
131  return median_label;
132 }
