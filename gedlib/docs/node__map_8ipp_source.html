174 void
175 NodeMap::
176 as_relation(std::vector<Assignment> & relation) const {
177  relation.clear();
178  GEDGraph::NodeID i, k;
179  for (i = 0; i < forward_map_.size(); i++) {
180  k = forward_map_.at(i);
181  if (k != GEDGraph::undefined_node()) {
182  relation.emplace_back(i, k);
183  }
184  }
185  for (k = 0; k < backward_map_.size(); k++) {
186  i = backward_map_.at(k);
187  if (i == GEDGraph::dummy_node()) {
188  relation.emplace_back(i, k);
189  }
190  }
