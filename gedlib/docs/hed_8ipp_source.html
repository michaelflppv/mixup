49 ged_run_(const GEDGraph & g, const GEDGraph & h, Result & result) {
50  DMatrix lsape_instance(g.num_nodes() + 1, h.num_nodes() + 1);
51  populate_instance_(g, h, lsape_instance);
52  double hed{0};

127 template<class UserNodeLabel, class UserEdgeLabel>
128 void
129 HED<UserNodeLabel, UserEdgeLabel>::
130 populate_instance_(const GEDGraph & g, const GEDGraph & h, DMatrix & lsape_instance) const {
131 
132 #ifdef _OPENMP
133  omp_set_num_threads(this->num_threads_ - 1);
134 #pragma omp parallel for if(this->num_threads_ > 1)
135 #endif
136  for (std::size_t row_in_master = 0; row_in_master < lsape_instance.num_rows(); row_in_master++) {
137  for (std::size_t col_in_master = 0; col_in_master < lsape_instance.num_cols(); col_in_master++) {
138  if ((row_in_master < g.num_nodes()) and (col_in_master < h.num_nodes())) {
139  lsape_instance(row_in_master, col_in_master) = compute_substitution_cost_(g, h, row_in_master, col_in_master);
140  }
141  else if (row_in_master < g.num_nodes()) {
142  lsape_instance(row_in_master, h.num_nodes()) = compute_deletion_cost_(g, row_in_master);
143  }
144  else if (col_in_master < h.num_nodes()) {
145  lsape_instance(g.num_nodes(), col_in_master) = compute_insertion_cost_(h, col_in_master);
146  }
147  }
148  }
149 }

151 template<class UserNodeLabel, class UserEdgeLabel>
152 double
153 HED<UserNodeLabel, UserEdgeLabel>::
154 compute_substitution_cost_(const GEDGraph & g, const GEDGraph & h, GEDGraph::NodeID i, GEDGraph::NodeID k) const {
155  // Collect node substitution costs.
156  double cost{this->ged_data_.node_cost(g.get_node_label(i), h.get_node_label(k))};
157 
158  // Initialize subproblem.
159  DMatrix subproblem(g.degree(i) + 1, h.degree(k) + 1);
160 
161  // Collect edge deletion costs.
162  std::size_t j{0};
163  for (auto ij = g.incident_edges(i).first; ij != g.incident_edges(i).second; ij++, j++) {
164  subproblem(j, h.degree(k)) = this->ged_data_.edge_cost(g.get_edge_label(*ij), ged::dummy_label()) / 2.0;
165  }
166 
167  // Collect edge insertion costs.
168  std::size_t l{0};
169  for (auto kl = h.incident_edges(k).first; kl != h.incident_edges(k).second; kl++, l++) {
170  subproblem(g.degree(i), l) = this->ged_data_.edge_cost(ged::dummy_label(), h.get_edge_label(*kl)) / 2.0;
171  }
172  j = 0;
173 
174  // Collect edge relabelling costs.
175  for (auto ij = g.incident_edges(i).first; ij != g.incident_edges(i).second; ij++, j++) {
176  l = 0;
177  for (auto kl = h.incident_edges(k).first; kl != h.incident_edges(k).second; kl++, l++) {
178  subproblem(j, l) = this->ged_data_.edge_cost(g.get_edge_label(*ij), h.get_edge_label(*kl)) / 2.0;
179  }
180  if (use_hed_for_edge_set_distances_) {
181  subproblem(j, l) /= 2.0;
182  }
183  }
184 
185  // Solve subproblem and update overall substitution costs.
186  if (use_hed_for_edge_set_distances_) {
187  cost += subproblem.matrix().block(0, 0, g.degree(i), h.degree(k) + 1).rowwise().minCoeff().sum();
188  cost += subproblem.matrix().block(0, 0, g.degree(i) + 1, h.degree(k)).colwise().minCoeff().sum();
189  }
190  else {
191  LSAPESolver subproblem_solver(&subproblem);
192  subproblem_solver.set_model(this->lsape_model_);
193  subproblem_solver.solve();
194  cost += subproblem_solver.minimal_cost();
195  }
196 
197  // Return the substitution costs divided by 2.
198  return cost / 2.0;
199 }
