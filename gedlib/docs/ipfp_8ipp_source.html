51 void
52 IPFP<UserNodeLabel, UserEdgeLabel>::
53 ls_run_from_initial_solution_(const GEDGraph & g, const GEDGraph & h, double lower_bound, const NodeMap & initial_node_map, NodeMap & output_node_map) {
54 
55  // Start timer.
56  Timer timer(time_limit_in_sec_);
57 
58  // Initialize output node map and upper bound.
59  output_node_map = initial_node_map;
60  double upper_bound{output_node_map.induced_cost()};
61 
62  // Initialize fractional solution x (is integral at start).
63  DMatrix x;
64  node_map_to_matrix_(initial_node_map, x);
65  double linear_cost_x{compute_induced_linear_cost_(qap_instance_, x)};
66  double overall_cost_x{initial_node_map.induced_cost()};
67  bool x_is_integral{true};
68 
69  // Initialize gradient direction b.
70  DMatrix b(qap_instance_.num_rows(), qap_instance_.num_cols());
71  double linear_cost_b{0.0};
72  double overall_cost_b{0.0};
73 
74  // Initialize alpha, beta, and the step width.
75  double alpha{0.0};
76  double beta{0.0};
77  double step_width{0.0};
78 
79  // Initialize linear problem and solvers.
80  DMatrix linear_problem(qap_instance_.num_rows(), qap_instance_.num_cols());
81  LSAPESolver lsape_solver;
82  lsape_solver.set_model(lsape_model_);
83  LSAPSolver lsap_solver;
84  if (quadratic_model_ == QAPE) {
85  lsape_solver.set_problem(&linear_problem);
86  }
87  else {
88  lsap_solver.set_problem(&linear_problem);
89  }
90  double min_linear_problem{0.0};
91 
92  // Initialize linear cost matrix.
93  DMatrix linear_cost_matrix(qap_instance_.num_rows(), qap_instance_.num_cols());
94  init_linear_cost_matrix_(qap_instance_, linear_cost_matrix);
95 
96 
97  // Main loop.
98  for (std::size_t current_itr{1}; not termination_criterion_met_(timer, alpha, min_linear_problem, current_itr, lower_bound, upper_bound); current_itr++) {
99 
100 
101  // Compute the next gradient direction b and update the upper bound and the output node map.
102  init_next_linear_problem_(qap_instance_, x, linear_cost_matrix, linear_problem);
103  if (quadratic_model_ == QAPE) {
104  solve_linear_problem_(qap_instance_, lsape_solver, min_linear_problem, linear_cost_b, overall_cost_b, b);
105  if (overall_cost_b < upper_bound) {
106  upper_bound = overall_cost_b;
107  util::construct_node_map_from_solver(lsape_solver, output_node_map);
108  this->ged_data_.compute_induced_cost(g, h, output_node_map);
109  if (std::fabs(upper_bound - output_node_map.induced_cost()) > 0.000001) {
110  throw Error("upper_bound: " + std::to_string(upper_bound) + ", linear_cost_b: " + std::to_string(linear_cost_b) + ", quadratic_cost_b: " + std::to_string(2*(upper_bound - linear_cost_b)) + ". Induced cost: " + std::to_string(output_node_map.induced_cost()) + ".");
111  }
112  output_node_map.set_induced_cost(upper_bound);
113  }
114  }
115  else {
116  solve_linear_problem_(qap_instance_, lsap_solver, min_linear_problem, linear_cost_b, overall_cost_b, b);
117  if (overall_cost_b < upper_bound) {
118  upper_bound = overall_cost_b;
119  util::construct_node_map_from_solver(lsap_solver, output_node_map);
120  output_node_map.set_induced_cost(upper_bound);
121  }
122  }
123 
124  // Determine the step width.
125  alpha = min_linear_problem - 2 * overall_cost_x + linear_cost_x;
126  beta = overall_cost_b + overall_cost_x - min_linear_problem - linear_cost_x;
127  if (beta > 0.00001) {
128  step_width = -alpha / (2 * beta);
129  }
130 
131  // Update the possibly fractional solution x.
132  if (beta <= 0.00001 or step_width >= 1) {
133  x_is_integral = true;
134  x = b;
135  overall_cost_x = overall_cost_b;
136  linear_cost_x = linear_cost_b;
137  }
138  else {
139  x_is_integral = false;
140  x += (b - x) * step_width;
141  overall_cost_x -= (alpha * alpha) / (4 * beta);
142  linear_cost_x = compute_induced_linear_cost_(qap_instance_, x);
143  }
144  }
145 
146 
147  // If converged solution x is fractional, project it to integral solution.
148  if (not x_is_integral) {
149  DMatrix projection_problem(qap_instance_.num_rows(), qap_instance_.num_cols(), 1.0);
150  projection_problem -= x;
151  NodeMap projected_node_map(output_node_map);
152  if (quadratic_model_ == QAPE) {
153  lsape_solver.set_problem(&projection_problem);
154  lsape_solver.solve();
155  util::construct_node_map_from_solver(lsape_solver, projected_node_map);
156  }
157  else {
158  lsap_solver.set_problem(&projection_problem);
159  lsap_solver.solve();
160  util::construct_node_map_from_solver(lsap_solver, projected_node_map);
161  }
162  this->ged_data_.compute_induced_cost(g, h, projected_node_map);
163  if (projected_node_map.induced_cost() < output_node_map.induced_cost()) {
164  output_node_map = projected_node_map;
165  }
166  }
167 }

276 template<class UserNodeLabel, class UserEdgeLabel>
277 void
278 IPFP<UserNodeLabel, UserEdgeLabel>::
279 node_map_to_matrix_(const NodeMap & node_map, DMatrix & matrix) const {
280  matrix.resize(qap_instance_.num_rows(), qap_instance_.num_cols());
281  matrix.set_to_val(0.0);
282  std::vector<NodeMap::Assignment> relation;
283  node_map.as_relation(relation);
284  for (auto assignment : relation) {
285  std::size_t row{undefined()};
286  std::size_t col{undefined()};
287  if (assignment.first != GEDGraph::dummy_node()) {
288  row = assignment.first;
289  }
290  if (assignment.second != GEDGraph::dummy_node()) {
291  col = assignment.second;
292  }
293  if ((row != undefined()) and (col != undefined())) {
294  matrix(row, col) = 1.0;
295  }
296  else if (row != undefined()) {
297  if (quadratic_model_ == QAPE) {
298  matrix(row, matrix.num_cols() - 1) = 1.0;
299  }
300  else if (quadratic_model_ == B_QAP){
301  matrix(row, row + qap_instance_.num_nodes_h()) = 1.0;
302  }
303  }
304  else if (col != undefined()) {
305  if (quadratic_model_ == QAPE) {
306  matrix(matrix.num_rows() - 1, col) = 1.0;
307  }
308  else if (quadratic_model_ == B_QAP){
309  matrix(col + qap_instance_.num_nodes_g(), col) = 1.0;
310  }
311  }
312  }
313 }

327 template<class UserNodeLabel, class UserEdgeLabel>
328 void
329 IPFP<UserNodeLabel, UserEdgeLabel>::
330 init_next_linear_problem_(const QAPInstance_ & qap_instance, const DMatrix & x, const DMatrix & linear_cost_matrix, DMatrix & linear_problem) const {
331  std::vector<NodeMap::Assignment> support_x;
332  for (std::size_t col{0}; col < linear_problem.num_cols(); col++) {
333  for (std::size_t row{0}; row < linear_problem.num_rows(); row++) {
334  if (x(row, col) != 0) {
335  support_x.emplace_back(row, col);
336  }
337  }
338  }
339  for (std::size_t col{0}; col < linear_problem.num_cols(); col++) {
340  for (std::size_t row{0}; row < linear_problem.num_rows(); row++) {
341  linear_problem(row, col) = 0.0;
342  for (const auto & assignment : support_x) {
343  linear_problem(row, col) += qap_instance(row, col, assignment.first, assignment.second) * x(assignment.first, assignment.second);
344  }
345  }
346  }
347  linear_problem += linear_cost_matrix;
348 }

350 template<class UserNodeLabel, class UserEdgeLabel>
351 bool
352 IPFP<UserNodeLabel, UserEdgeLabel>::
353 termination_criterion_met_(const Timer & timer, const double & alpha, const double & min_linear_problem, const std::size_t & current_itr, double lower_bound, double upper_bound) const {
354  if (current_itr == 1) {
355  return false;
356  }
357  if (lower_bound >= upper_bound) {
358  return true;
359  }
360  if (timer.expired()) {
361  return true;
362  }
363  if (max_itrs_ > 0 and current_itr > max_itrs_) {
364  return true;
365  }
366  if (min_linear_problem < 0.00001) {
367  return std::fabs(alpha) < epsilon_;
368  }
369  return std::fabs(alpha / min_linear_problem) < epsilon_;
370 }

373 void
374 IPFP<UserNodeLabel, UserEdgeLabel>::
375 solve_linear_problem_(const QAPInstance_ & qap_instance, LSAPSolver & solver,
376  double & min_linear_problem, double & linear_cost_b, double & overall_cost_b, DMatrix & b) const {
377  solver.solve();
378  min_linear_problem = solver.minimal_cost();
379  solver_to_matrix_(solver, b);

381  overall_cost_b = linear_cost_b + 0.5 * compute_induced_quadratic_cost_(qap_instance, solver);
382 }

468 compute_induced_quadratic_cost_(const QAPInstance_ & qap_instance, const LSAPSolver & solver) const {
469  double quadratic_cost{0.0};
470  std::vector<NodeMap::Assignment> assignments;
471  for (std::size_t row{0}; row < solver.num_rows(); row++) {
472  if (solver.get_assigned_col(row) < solver.num_cols()) {
473  assignments.emplace_back(row, solver.get_assigned_col(row));
474  }
475  }
476  for (const auto & assignment_1 : assignments) {
477  for (const auto & assignment_2 : assignments) {
478  quadratic_cost += qap_instance(assignment_1.first, assignment_1.second, assignment_2.first, assignment_2.second);
479  }
480  }
481  return quadratic_cost;
482 }
