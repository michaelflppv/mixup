102  if (num_solutions > 1) {

113  row_to_col_assignments_.push_back(std::vector<std::size_t>(num_rows()));
114  for (int row{0}; row < static_cast<int>(num_rows()); row++) {
115  row_to_col_assignments_.back().at(row) = matching(0, row);
116  }
117  col_to_row_assignments_.push_back(std::vector<std::size_t>(num_cols()));

226 std::vector<std::size_t>
227 LSAPSolver ::
228 construct_col_to_row_asignment_(const std::vector<std::size_t> row_to_col_assignment) const {
229  std::vector<std::size_t> col_to_row_assignment(num_cols(), num_rows());
230  for (std::size_t row{}; row < num_rows(); row++) {
231  if (row_to_col_assignment.at(row) < num_cols()) {
232  col_to_row_assignment[row_to_col_assignment.at(row)] = row;
233  }
234  }
235  return col_to_row_assignment;
236 }

238 void
239 LSAPSolver ::
240 compute_cost_from_assignments_() {
241  minimal_cost_ = 0.0;
242  for (std::size_t row{}; row < num_rows(); row++) {
243  minimal_cost_ += cost_matrix_->operator()(row, row_to_col_assignments_.at(0).at(row));
244  }
245  for (std::size_t col{}; col < num_cols(); col++) {
246  if (col_to_row_assignments_.at(0).at(col) == num_rows()) {
247  minimal_cost_ += cost_matrix_->operator()(num_rows(), col);
248  }
249  }
250 }

260  minimal_cost_ += *dual_var;
261  }
